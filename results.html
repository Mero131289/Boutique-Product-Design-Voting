'selected' : ''}`; card.innerHTML = ` `; $('button', card).addEventListener('click', () => onSelect(option.id)); return card; } function renderTopbar(root, state, extraRight = '') { const topbar = $('.topbar', root); if (!topbar) return; topbar.innerHTML = `
${icon('users')} ${state.capacity} guests
${extraRight || `
${icon('shield')} Anonymous
`} `; } function formatPct(value) { return `${Number(value || 0).toFixed(1).replace('.0', '')}%`; } function winnerFromState(state) { if (!state.options.length) return null; const sorted = [...state.options].sort((a, b) => (state.counts[b.id] || 0) - (state.counts[a.id] || 0)); return sorted[0]; } function renderResultsRows(container, state) { container.innerHTML = ''; state.options.forEach((option) => { const votes = state.counts[option.id] || 0; const pct = state.percentages[option.id] || 0; const row = document.createElement('div'); row.className = 'result-row'; row.innerHTML = `
${option.label}${formatPct(pct)}
${votes}votes
`; container.appendChild(row); }); } function downloadCsv(state) { const rows = [['Option', 'Votes', 'Percentage']]; state.options.forEach((option) => rows.push([ option.label, state.counts[option.id] || 0, `${state.percentages[option.id] || 0}%` ])); rows.push(['Total', state.total, '100%']); const csv = rows.map(row => row.map(value => `"${String(value).replaceAll('"', '""')}"`).join(',')).join('\n'); const blob = new Blob([csv], { type: 'text/csv;charset=utf-8' }); const link = document.createElement('a'); link.href = URL.createObjectURL(blob); link.download = `favorite-design-poll-results-${new Date().toISOString().slice(0, 10)}.csv`; link.click(); URL.revokeObjectURL(link.href); } function makeAnonymousKey(pollId) { const keyName = `favorite-design-voter-key-${pollId}`; let key = localStorage.getItem(keyName); if (!key) { const raw = crypto.randomUUID ? crypto.randomUUID() : `${Date.now()}-${Math.random().toString(36).slice(2)}`; key = raw.replace(/[^a-zA-Z0-9_-]/g, '').slice(0, 80); localStorage.setItem(keyName, key); } return key; } window.PollApp = { $, $$, getState, connectEvents, icon, optionCard, renderTopbar, formatPct, winnerFromState, renderResultsRows, downloadCsv, makeAnonymousKey };
